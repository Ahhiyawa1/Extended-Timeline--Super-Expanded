subject_has_overlord_color_effect = {
	if = {
		limit = {
			is_subject_other_than_tributary_trigger = yes
			OR = {
				is_subject_of_type = swiss_canton_subject
				is_subject_of_type = substate_subject
				is_subject_of_type = crown_dependency_subject
				#is_subject_of_type = governorate_subject
				AND = {
					always = no # More triggers here
				}
				AND = {
					has_global_flag = CONFIG_subject_has_overlord_color_global_flag
					NOT = { is_colonial_nation = yes }
				}
			}
		}
		change_country_color = { country = overlord }
		set_country_flag = has_overlord_color
	}
	else_if = {
		limit = {
			has_country_flag = has_overlord_color
		}
		restore_country_color = yes
		clr_country_flag = has_overlord_color
	}
}

save_as_underscore_global_event_target_effect = {
	$tag$ = {
		save_global_event_target_as = _$tag$ # Ex. "'GBR' saved as 'event_target:_GBR'"
		#log = "'$tag$' saved as 'event_target:_$tag$'"
	}
}

update_country_tag_alias_list = {
	effect_with_tag_parameter = {
		scripted_effect = save_as_underscore_global_event_target_effect
	}
	update_country_tag_alias = {
		define_mode = yes # Saves alias-tags as Global Flags/Event targets
	}
}

define_country_tag_alias = {
	if = {
		limit = {
			has_global_flag = update_country_tag_alias_define_mode
		}
		[[variant]
			set_global_flag = $default$_alias:$variant$ # Define '$variant$' as an alias for '$default$' tag
		]
	}
	else_if = {
		limit = {
			country_is = { tag = $default$ }
		}
		if = {
			limit = {
				[[trigger]
					$trigger$ # Thanks to the 'Ante Bellum' mod for teaching me about this neat trick.
				]
			}
			if = {
				limit = {
					NOT = { has_country_flag = tag_has_changed_flag }
				}
				[[variant]
					if = {
						limit = {
							exists = yes
							NOT = { tag = $variant$ }
							NOT = { exists = $variant$ }
						}
						trigger_switch = {
							on_trigger = tag
							event_target:GP_1  = { $variant$ = { save_global_event_target_as = GP_1  } }
							event_target:GP_2  = { $variant$ = { save_global_event_target_as = GP_2  } }
							event_target:GP_3  = { $variant$ = { save_global_event_target_as = GP_3  } }
							event_target:GP_4  = { $variant$ = { save_global_event_target_as = GP_4  } }
							event_target:GP_5  = { $variant$ = { save_global_event_target_as = GP_5  } }
							event_target:GP_6  = { $variant$ = { save_global_event_target_as = GP_6  } }
							event_target:GP_7  = { $variant$ = { save_global_event_target_as = GP_7  } }
							event_target:GP_8  = { $variant$ = { save_global_event_target_as = GP_8  } }
							event_target:GP_9  = { $variant$ = { save_global_event_target_as = GP_9  } }
							event_target:GP_10 = { $variant$ = { save_global_event_target_as = GP_10 } }
							event_target:GP_11 = { $variant$ = { save_global_event_target_as = GP_11 } }
							event_target:GP_12 = { $variant$ = { save_global_event_target_as = GP_12 } }
							event_target:GP_13 = { $variant$ = { save_global_event_target_as = GP_13 } }
							event_target:GP_14 = { $variant$ = { save_global_event_target_as = GP_14 } }
							event_target:GP_15 = { $variant$ = { save_global_event_target_as = GP_15 } }
							event_target:GP_16 = { $variant$ = { save_global_event_target_as = GP_16 } }
						}
						save_event_target_as = prev_tag
						
						change_tag = $variant$
						every_core_province = {
							limit = {
								NOT = { owned_by = $variant$ }
							}
							add_core = $variant$ # Refresh foreign cores, they expire too quickly sometimes.
						}
						if = {
							limit = {
								has_saved_event_target = prev_tag
							}
							every_country = {
								limit = {
									historical_friend_with = event_target:prev_tag
								}
								mutual_add_historical_friend = { who = $variant$ }
							}
							every_country = {
								limit = {
									historical_rival_with = event_target:prev_tag
								}
								mutual_add_historical_rival = { who = $variant$ }
							}
						}
					}
				]
				set_country_flag = tag_has_changed_flag # This indicates that we've found a variant that fulfils the triggers.
				
				# Optional effect if true:
				[[effect]
					$effect$
				]
				# Optional color change:
				[[country_color]
					if = {
						limit = {
							NOT = { has_country_flag = color_has_changed_flag }
						}
						change_country_color = { $country_color$ } # country_color = "country = <country scope>" | "color = { r g b }"
						clr_country_flag = has_overlord_color
						set_country_flag = color_has_changed_flag
						
						# every_subject_country = {
							# limit = {
								# is_colonial_nation = yes
							# }
							# change_country_color = { country = overlord }
						# }
					}
				]
			}
		}
		else = {
			# Optional effect if untrue:
			[[removed_effect]
				$removed_effect$
			]
			# Restore color change:
			[[country_color]
				if = {
					limit = {
						NOT = { has_country_flag = color_has_changed_flag }
					}
					restore_country_color = yes
					
					# every_subject_country = {
						# limit = {
							# is_colonial_nation = yes
						# }
						# change_country_color = { country = overlord }
					# }
				}
			]
		}
	}
	[[variant]
		$variant$ = { # Event targets:
			if = {
				limit = {
					exists = yes
					OR = {
						NOT = { has_saved_global_event_target = _$default$ }
						AND = {
							NOT = { tag = event_target:_$default$ }
							
							event_target:_$default$ = {
								OR = {
									exists = no
									NOT = { total_development = $variant$ }
								}
							}
						}
					}
				}
				save_global_event_target_as = _$default$ # Ex. "'UKI' saved as 'event_target:_GBR'"
				log = "'$variant$' saved as 'event_target:_$default$'"
			}
		}
	]
}
