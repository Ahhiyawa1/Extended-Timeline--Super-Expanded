date_tick = {
    export_to_variable = { which = year value = trigger_value:is_year }
    change_variable = { which = year value = -2099 }
	if = {
		limit = {
			NOT = { check_variable = { which = year value = 0 } }
		}
		set_variable = { which = year_bc which = year }
		multiply_variable = { which = year_bc value = -1 }
	}
	else = {
		set_variable = { which = year_bc value = 0 }
	}
}
