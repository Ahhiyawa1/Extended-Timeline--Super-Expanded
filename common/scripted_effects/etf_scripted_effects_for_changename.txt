override_country_name_add_flag = {
	if = {
		limit = {
			NOT = { has_country_flag = has_special_name_flag }
		}
		# Clear out existing names/flags
		restore_country_name_all = yes
		# Add name alongside matching flag
		override_country_name = $name$
		set_country_flag = is_$name$_flag
		# Optional:
		[[definite]
			if = {
				limit = {
					always = $definite$ # definite = yes|no
				}
				set_country_flag = definite_name_flag # Adds "the" when name is used in localization
			}
			else = {
				set_country_flag = not_definite_name_flag # Override. Removes "the" when used in localization
			}
		]
		[[special]
			# With this flag set, another name cannot easily overwrite this name.
			# The name would have to be manually removed with 'clear_country_name_with_flag' first,
			# which clears this flag.
			if = {
				limit = {
					always = $special$ # special = yes
				}
				set_country_flag = has_special_name_flag
			}
		]
	}
}

clear_country_name_with_flag = {
	if = {
		limit = {
			has_country_flag = is_$name$_flag
		}
		restore_country_name = yes
		clr_country_flag = is_$name$_flag
		clr_country_flag = definite_name_flag
		clr_country_flag = not_definite_name_flag
		clr_country_flag = has_special_name_flag
	}
}

dynamic_country_name = { # $name$, $definite$, $trigger$
	if = {
		limit = {
			NOT = { has_country_flag = name_is_being_changed }
			[[trigger]
				$trigger$
			]
		}
		override_country_name_add_flag = {
			name = $name$
			[[definite] definite = $definite$]
		}
		set_country_flag = name_is_being_changed # Once a correct name has been found, stop searching
	}
}
