					}

				}

						has_reform = responsible_government_reform
					}

						subject = ROOT
						subject_type = dominion
					}
					if = {
						limit = {
							overlord = {
								government = monarchy
							}
						}
						change_government = monarchy
						add_government_reform = dominion_reform
					}
				}
				else = {
					# Sovereign Dominion
					if = {
						limit = {
							is_subject = yes
						}
						grant_independence = yes
					}
					if = {
						limit = {
							has_reform = dominion_reform
						}
						add_government_reform = sovereign_dominion_reform
					}
				}
			}
			ai_chance = {
				factor = 1
			}
		}
	}
