	chn_escape_to_taiwan = {
		major = yes
		potential = {
			country_is = { tag = CHN }
			OR = {
				war_with = PRC
				AND = {
					has_spawned_rebels = communist_rebels
					NOT = { exists = PRC }
				}
			}
			taiwan_area = {
				type = all
				is_core 	  = ROOT
				owned_by 	  = ROOT
				controlled_by = ROOT
			}
			any_owned_province = {
				NOT = { area = taiwan_area }
			}
			NOT = { is_communist = yes }
			NOT = { has_country_flag = flight_to_taiwan_flag }
		}
		allow = {
			OR = {
				if = {
					limit = { war_with = PRC }
					war_score_against = {
						who = PRC
						value = -50
					}
				}
				num_of_owned_provinces_with = {
					value = 30
					has_rebel_faction = communist_rebels
				}
			}
		}
		effect = {
			set_capital = 2154
			every_subject_country = {
				grant_independence = yes
			}
			custom_tooltip = chn_escape_to_taiwan_tt
			hidden_effect = {
				disband_rebels = communist_rebels
				every_core_province = {
					if = {
						limit = { area = taiwan_area }
						remove_core = PRC
					}
					else = {
						remove_core = ROOT
						add_core = PRC
					}
				}
				# Release the PRC
				if = {
					limit = {
						exists = PRC
					}
					white_peace = PRC
					every_owned_province = {
						limit = {
							NOT = { area = taiwan_area }
						}
						cede_province = PRC
					}
				}
				else = {
					release = PRC
					PRC = {
						change_government_to_dictatorial_republic = yes
						add_government_reform = communist_government
						add_treasury = 250
						add_manpower = 50
						capital_scope = {
							build_to_forcelimit = {
								infantry = 0.7
								cavalry = 0.1
								artillery = 0.2
							}
						}
					}
				}
				# Add post-setup claims/cores
				every_province = {
					limit = {
						is_core = PRC
					}
					add_permanent_claim = ROOT
				}
				taiwan_area = {
					add_core = PRC
				}
			}
			change_tag = TWA
			on_change_tag_effect = yes
			set_country_flag = flight_to_taiwan_flag
		}
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				NOT = {
					OR = {
						capital_scope = {
							controlled_by = REB
						}
						num_of_owned_provinces_with = {
							value = 80
							has_rebel_faction = communist_rebels
						}
					}
				}
			}
		}
	}
