country_event = {
	id = cys.2
	title = "cys1_title"
	picture = "RELIGIOUS_WARS_eventPicture"
	desc = "cys1_desc"
	is_triggered_only = yes
	hidden = yes
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					has_global_flag = include_every_subject
				}
				every_subject_country = {
					change_country_color = {
						country = PREV
					}
					if = {
						limit = {
							num_of_subjects = 1
						}
						every_subject_country = {
							change_country_color = {
								country = PREV
							}
						}
					}
				}
			}
			else_if = {
				limit = {
					has_global_flag = exclude_shogunate_vassal
				}
				every_subject_country = {
					limit = {
						NOT = {
							is_subject_of_type = daimyo_vassal
						}
					}
					change_country_color = {
						country = PREV
					}
					if = {
						limit = {
							num_of_subjects = 1
						}
						every_subject_country = {
							limit = {
								NOT = {
									is_subject_of_type = daimyo_vassal
								}
							}
							change_country_color = {
								country = PREV
							}
						}
					}
				}
			}
			else_if = {
				limit = {
					has_global_flag = exclude_tributary_subject
				}
				every_subject_country = {
					limit = {
						NOT = {
							is_subject_of_type = tributary_state
						}
					}
					change_country_color = {
						country = PREV
					}
					if = {
						limit = {
							num_of_subjects = 1
						}
						every_subject_country = {
							limit = {
								NOT = {
									is_subject_of_type = tributary_state
								}
							}
							change_country_color = {
								country = PREV
							}
						}
					}
				}
			}
			else_if = {
				limit = {
					has_global_flag = exclude_shogunate_tributary_subjects
				}
				every_subject_country = {
					limit = {
						NOT = {
							OR = {
								is_subject_of_type = daimyo_vassal
								is_subject_of_type = tributary_state
							}
						}
					}
					change_country_color = {
						country = PREV
					}
					if = {
						limit = {
							num_of_subjects = 1
						}
						every_subject_country = {
							limit = {
								NOT = {
									OR = {
										is_subject_of_type = daimyo_vassal
										is_subject_of_type = tributary_state
									}
								}
							}
							change_country_color = {
								country = PREV
							}
						}
					}
				}
			}
		}
	}
	option = {
		name = cys.1a
	}
}
