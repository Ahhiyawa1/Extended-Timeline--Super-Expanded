# UPCA Breaks Apart
country_event = {
	id = etf_event_cam.130
	title = etf_event_cam.130.t
	desc = etf_event_cam.130.d
	picture = BORDER_TENSION_eventPicture
	major = yes
	major_trigger = {
		capital_scope = {
			region = central_america_region
		}
	}

	fire_only_once = yes

	trigger = {
		has_country_flag = central_american_civil_war
		is_at_war = no
		any_country = {
			has_country_flag = central_american_breakaway
			NOT = { is_subject_of = ROOT }
		}
	}

	immediate = {
		hidden_effect = {
			clr_country_flag = central_american_civil_war
		}
	}

	option = {
		name = etf_event_cam.130.r
		change_tag = GTM
		on_change_tag_effect = yes
		if = {
			limit = {
				government_rank = 3
			}
			set_government_rank = 2
		}
		if = {
			limit = {
				owns_core_province = 2653
			}
			set_capital = 2653
		}
		if = {
			limit = {
				has_custom_ideas = no
			}
			country_event = { id = ideagroups.1 } #Swap Ideas
		}
		set_country_flag = changed_from_colonial_nation
		hidden_effect = {
			# El Salvador leaves as well
			if = {
				limit = {
					any_owned_province = {
						is_core = ELS
					}
					NOT = { exists = ELS }
				}
				start_civil_war = { # Easy way to release
					tag = ELS
					government = republic
					treasury = 50
					manpower = 5
				}
				ELS = {
					set_country_flag = changed_from_colonial_nation
				}
			}
		}
	}

	after = {
		every_country = {
			limit = {
				has_country_flag = central_american_breakaway
			}
			clr_country_flag = central_american_breakaway
		}
		every_core_province = {
			limit = {
				OR = {
					is_core = NCR
					is_core = HON
					is_core = COS
					is_core = ELS
				}
				NOT = { owned_by = ROOT }
			}
			remove_core = ROOT
		}
	}
}
