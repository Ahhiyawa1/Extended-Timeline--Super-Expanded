					is_free_or_tributary_trigger = yes
					NOT = { war_with = ROOT }
					NOT = { is_democracy = yes }
					NOT = { is_communist = yes }
				}
				FROM = {
					owns = 1952
				}
			}
			event_target:_HUN = {
				country_event = { id = etf_event_hoi.233 }
			}
		}
		tooltip = {
			if = {
				limit = {
					FROM = {
						any_owned_province = {
							culture = slovak
						}
					}
					NOT = { exists = SVK }
				}
				create_subject = {
					subject = SVK
					subject_type = puppet_state
				}
			}
		}
		hidden_effect = {
			FROM = {
				if = {
					limit = {
						any_owned_province = {
							culture = slovak
						}
						NOT = { exists = SVK }
					}
					every_owned_province = {
						limit = {
							culture = slovak
						}
						add_core = SVK
					}
					release = SVK
					SVK = {
						change_government_to_dictatorial_republic = yes
						add_government_reform = fascist_government
					}
					ROOT = {
						create_subject = {
							subject = SVK
							subject_type = puppet_state
						}
					}
				}
			}
		}
		annex_tag = {
			target = FROM
			transfer_troops = no
		}
	}

	after = {
		if = {
			limit = {
				event_target:_POL = {
					is_free_or_tributary_trigger = yes
					any_owned_province = {
						is_core = ROOT
					}
					NOT = { war_with = ROOT }
					NOT = { alliance_with = ROOT }
					
					event_target:_GBR = {
						is_great_power = yes
						NOT = { war_with = PREV }
					}
				}
			}
			event_target:_POL = {
				country_event = { id = etf_event_hoi.240 days = 5 }
			}
		}
	}
}
