namespace = etf_event_arb

# The Omani Succession crisis
country_event = {
	id = etf_event_arb.100
	title = etf_event_arb.100.t
	desc = etf_event_arb.100.d
	picture = COUNTRY_COLLAPSE_eventPicture

	fire_only_once = yes

	trigger = {
		country_is = { tag = OMA }
		is_at_war = no
		owns_or_non_sovereign_subject_of = 400  # Muscat
		owns_or_non_sovereign_subject_of = 1201 # Zanzibar
		# Wouldn't want players randomly getting annexed without warning:
		NOT = {
			any_subject_country = {
				ai = no
				capital_scope = {
					OR = {
						region = horn_of_africa_region
						region = central_africa_region
						region = east_africa_region
					}
				}
			}
		}
		NOT = { exists = ZZB }
		
		is_year = 3954
		started_in = 3922.1.1
		NOT = { started_in = 3954.10.19 }
	}

	option = { # Our reign continues from Muscat.
		name = etf_event_arb.100.ra
		400 = {
			move_capital_effect = yes
		}
		if = {
			limit = {
				government_rank = 3
			}
			set_government_rank = 2
		}
		custom_tooltip = " "
		tooltip = {
			release = ZZB
		}
		hidden_effect = {
			horn_of_africa_region = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
				}
				remove_core = ROOT
				add_core = ZZB
				cede_province = ZZB
			}
			central_africa_region = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
				}
				remove_core = ROOT
				add_core = ZZB
				cede_province = ZZB
			}
			east_africa_region = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
				}
				remove_core = ROOT
				add_core = ZZB
				cede_province = ZZB
			}
			# Free Somali Tributaries:
			every_subject_country = {
				limit = {
					capital_scope = {
						OR = {
							region = horn_of_africa_region
							region = central_africa_region
							region = east_africa_region
						}
					}
				}
				grant_independence = yes
			}
		}
		ZZB = {
			change_government = ROOT
			change_primary_culture = ROOT
			change_religion = ROOT
			set_capital = 1201 # Zanzibar
			define_ruler = {
				name = "Majid"
				dynasty = "bin Said"
				claim = 100
			}
			add_treasury = 100
			add_manpower = 10
		}
	}

	option = { # We'll rule the Swahili coast!
		name = etf_event_arb.100.rb
		trigger = {
			ai = no
		}
		ai_chance = {
			factor = 0
		}
		1201 = {
			move_capital_effect = yes
		}
		if = {
			limit = {
				government_rank = 3
			}
			set_government_rank = 2
		}
		change_tag = ZZB
		on_change_tag_effect = yes
		hidden_effect = {
			remove_heir = { allow_new_heir = no }
		}
		define_ruler = {
			name = "Majid"
			dynasty = "bin Said"
			claim = 100
		}
		custom_tooltip = " "
		tooltip = {
			release = OMA
		}
		hidden_effect = {
			arabia_region = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
				}
				remove_core = ROOT
				add_core = OMA
				cede_province = OMA
			}
			persia_region = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
				}
				remove_core = ROOT
				add_core = OMA
				cede_province = OMA
			}
			khorasan_region = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
				}
				remove_core = ROOT
				add_core = OMA
				cede_province = OMA
			}
		}
		OMA = {
			change_government = ROOT
			change_primary_culture = ROOT
			change_religion = ROOT
			set_capital = 400 # Muscat
			define_ruler = {
				name = "Thuwaini"
				dynasty = "Al Said"
				claim = 100
			}
			add_treasury = 100
			add_manpower = 10
		}
	}
}
