country_event = {
	id = confederation_event.1
	title ="ndb_inherits_sdb_title"
	desc = "ndb_inherits_sdb_desc"

	picture = DIPLOMACY_eventPicture
	

	trigger = {
		AND = {
			tag = NDB
			alliance_with = SDB	
			SDB = {has_regency = yes}

			NOT = { has_country_flag = did_not_want_inherit_sdb }
		}
	}

	mean_time_to_happen = {
		days = 5
		modifier = {
			factor = 0.2
			stability = 1
		}	
	}

	option = {
		name ="INHERIT_SDB"
		ai_chance = { factor = 100 }
		add_stability = -1	
		create_union = SDB
		add_prestige = 25
	}
	
	option = {
		name = "NOT_INHERIT_SDB"
		add_prestige = -10
		set_country_flag = did_not_want_inherit_sdb
	}	
}

				primary_culture = hannoverian 
				primary_culture = pommeranian

							primary_culture = hessian
							primary_culture = saxon
							primary_culture = prussian
						}

					NOT = {tag = LIV}
					NOT = {tag = RIG}

				}

			factor = 1

country_event = {
	id = confederation_event.7
	title = "form_germany_historically_title"
	desc = "form_germany_historically_desc"
	
	picture = HRE_eventPicture
	
	trigger = {
		AND = {
			NOT = {exists = GER}
			ai = no
			OR = {
				tag = NDB
				tag = SDB
			}
			any_rival_country = {
				defensive_war_with = ROOT
			}
			war_score = 70
			NOT = {has_country_flag = did_not_want_become_ger }
		}
	}
	
	
	option = {
		name = "GERMANY_FORMED_HISTORICALLY"
		if = {
			limit = {
				AND = {
					tag = NDB
						OR = {
							alliance_with = SDB
							SDB = {
								is_subject_of = ROOT
							}
						}
					SDB = {
						ai = yes
					}
				}
			}
			inherit = SDB
		}
		if = {
			limit = {
				AND = {
					tag = SDB
					OR = {
						alliance_with = NDB
						NDB = {
							is_subject_of = ROOT
						}
					}
					NDB = {
						ai = yes
					}
				}
			}
			inherit = NDB
		}
		change_tag = GER
		emperor = {
			add_opinion = {
				who = GER
				modifier = opinion_left_empire
			}
		}
		set_in_empire = no
		emperor = {
			country_event = { id = hre_event.5 }
		}
		set_government_rank = 3
		north_german_region = {
				limit = {
					NOT = { owned_by = ROOT }
				}
				add_claim = GER
			}
		south_german_region = {
			limit = {
				NOT = { owned_by = ROOT }
			}
			add_claim = GER
		}
		add_country_modifier = {
			name = "centralization_modifier"
			duration = 7300
		}
		add_prestige = 25
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "GERMANY_FORMED_HISTORICALLY_NOT"
		set_country_flag = did_not_want_become_ger
	}
}
