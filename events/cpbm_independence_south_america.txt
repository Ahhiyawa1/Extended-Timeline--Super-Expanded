#Colombia
country_event = {
	id = cpbm_independence_south_america.1
	title = cpbm_independence_south_america.1.t
	desc = cpbm_independence_south_america.1.d
	picture = MILITARY_CAMP_eventPicture

	is_triggered_only = yes

	trigger = {
        is_year = 3814
    }	
	
	option = {
		name = cpbm_independence_south_america.1.a
		add_adm_power = -200
		add_dip_power = -200
		add_mil_power = -200
		every_owned_province = { 
			limit = { 
				OR = {
					area = bogota_area
					area = western_llanos
					area = popayan_area
					area = coquivacoa_area
					area = cordillera_occidental_area
					area = colombian_amazonas_area
				}  
            }
			cede_province = COL
			add_core = COL
			remove_core = ROOT 
		}
		vassalize = COL
		COL = { set_capital = 825 }
	}
	
	option = {
		name = cpbm_independence_south_america.1.b
		every_owned_province = { 
			limit = { 
				OR = {
					area = bogota_area
					area = western_llanos
					area = popayan_area
					area = coquivacoa_area
					area = cordillera_occidental_area
					area = colombian_amazonas_area
				}  
            }
			cede_province = COL
			add_core = COL
			remove_core = ROOT 
		}
		COL = { 
			every_neighbor_country = { add_truce_with = COL } 
			set_capital = 825
		}
	}
	
	option = {
		name = cpbm_independence_south_america.1.c
		trigger = { AND = { adm_power = 100 dip_power = 100 mil_power = 100 } }
		add_adm_power = -200
		add_dip_power = -200
		add_mil_power = -200
		add_years_of_income = -2
		every_owned_province = { 
			limit = { 
				OR = {
					area = bogota_area
					area = western_llanos
					area = popayan_area
					area = coquivacoa_area
					area = cordillera_occidental_area
					area = colombian_amazonas_area
				}  
            }
			add_unrest = 5
			add_core = COL
		}
	}
	
	option = {
		name = cpbm_independence_south_america.1.dd
		every_owned_province = { 
			limit = { 
				OR = {
					area = bogota_area
					area = western_llanos
					area = popayan_area
					area = coquivacoa_area
					area = cordillera_occidental_area
					area = colombian_amazonas_area
				}  
            }
			cede_province = COL
			add_core = COL
		}
		COL = { 
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
			set_capital = 825
		}
		hidden_effect = {
			COL = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.8
						cavalry = 0.2
					}
				}
			}
		}
	}
    
    after = { 825 = { every_core_country = { limit = { tag = COL } change_primary_culture = PREV change_religion = PREV change_government = ROOT } }
		COL = { every_province = { limit = { owned_by = ROOT is_in_capital_area = no NOT = { range = ROOT } trade_range = PREV } cede_province = PREV add_core = PREV }
		every_owned_province = { limit = { is_in_capital_area = no NOT = { range = PREV } trade_range = ROOT } cede_province = ROOT add_core = ROOT } }
		random_list = { 1 = { country_event = { id = cpbm_independence_trigger.1 } } 9 = { } }
	}
}
