#Any Subject Nation
country_event = {
	id = cpbm_independence.1
	title = cpbm_independence.1.t
	desc = cpbm_independence.1.d
	picture = STREET_SPEECH_eventPicture

	is_triggered_only = yes
	
	option = {
		name = cpbm_independence.1.a
        trigger = { 
            overlord = { reverse_has_opinion = { who = ROOT value = 100 } } 
            adm_power = 50
            adm_power = 50
            adm_power = 50
            treasury = 100
        }
		add_adm_power = -50
		add_dip_power = -50
		add_mil_power = -50
		add_treasury = -100
		overlord = {
            add_opinion = {
                who = ROOT
                modifier = offered_support_to_us
            }
        }
		overlord = { country_event = { id = cpbm_independence.2 } }
		ai_chance = { factor = 100 }
	}
	
	option = {
		name = cpbm_independence.1.b
		ai_chance = { factor = 2 }
	}
	
	option = {
		name = cpbm_independence.1.c
        trigger = { NOT = { has_opinion = { who = FROM value = 100 } } }
		overlord = { country_event = { id = cpbm_independence.3 } }
		ai_chance = { factor = 3 }
	}
	
	option = {
		name = cpbm_independence.1.dd
        trigger = { NOT = { has_opinion = { who = FROM value = 0 } } }
		overlord = { country_event = { id = cpbm_independence.4 } }
		ai_chance = { factor = 4 }
	}
    
    after = { overlord = { random_list = { 1 = { country_event = { id = cpbm_independence_trigger.1 } } 9 = { } } } }
}

#Subject assists us
country_event = {
	id = cpbm_independence.2
	title = cpbm_independence.2.t
	desc = cpbm_independence.2.d
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes
	
	option = {
		name = cpbm_independence.2.a
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
		add_treasury = 100
		ai_chance = { factor = 4 }
	}
	
	option = {
		name = cpbm_independence.2.b
        trigger = { 
            has_opinion = { who = FROM value = 100 }
            trust = { who = FROM value = 50 }
        }
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
		add_treasury = 100
		FROM = {
            add_opinion = {
                who = ROOT
                modifier = released_us_freely
            }
        }
		FROM = { 
            grant_independence = yes
            country_event = { id = cpbm_independence.5 } 
        }
        create_alliance = FROM
        add_historical_friend = FROM
		ai_chance = { factor = 1 }
	}
}

            factor = 1
			modifier = { 
				factor = 3

                trust = { who = FROM value = 75 }
			}
        }
	}
}

#Subject declares war of independence
country_event = {
	id = cpbm_independence.4
	title = cpbm_independence.4.t
	desc = cpbm_independence.4.d
	picture = CIVIL_WAR_eventPicture

	is_triggered_only = yes
	
	option = {
		name = cpbm_independence.4.a
        FROM = {
            declare_war_with_cb = {
                who = ROOT
                casus_belli = cb_independence_war
            }
        }
        hidden_effect = {
			FROM = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.8
						cavalry = 0.2
					}
				}
			}
        }
	}
}
