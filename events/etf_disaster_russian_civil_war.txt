# Bolshevik Coup
country_event = {
	id = etf_event_rcw.1
	title = etf_event_rcw.1.t
	desc = etf_event_rcw.1.d
	picture = FEB_1917_eventPicture # Technically wrong revolution, but whatever.
	major = yes
	major_trigger = {
		OR = {
			same_continent = FROM
			is_neighbor_of = FROM
		}
	}

	is_triggered_only = yes

	immediate = {
		set_global_flag = russian_civil_war_flag
	}

	option = { # The Whites will restore order
		name = etf_event_rcw.1.ra
		russia_region = {
			limit = { owned_by = ROOT }
			add_core = RSF
		}
		start_civil_war = {
			tag = RSF
			treasury = 500
			manpower = 100
			religion = secularism
			government = republic
			communist = yes
			capital = 33 # Petrograd
			delete_cores_after = yes
			type = mutual
		}
	}

	option = { # The Bolsheviks know how to rule Russia
		name = etf_event_rcw.1.rb
		trigger = {
			ai = no
		}
		ai_chance = {
			factor = 0
		}
		russia_region = {
			limit = { owned_by = ROOT }
			add_core = RSF
		}
		start_civil_war = {
			tag = RSF
			treasury = 500
			manpower = 100
			religion = secularism
			government = republic
			communist = yes
			capital = 33
			delete_cores_after = yes
			type = mutual
			play_as = yes
		}
	}

	after = {
		RSF = {
			define_ruler = {
				name = "Vladimir Lenin"
				adm = 4
				dip = 2
				mil = 3
			}
			define_general = {
				name = "Leon Trotsky"
				shock = 4
				fire = 5
				manuever = 5
				siege = 4
			}
		}
	}
}

# Breakaway Region Secedes
province_event = {
	id = etf_event_rcw.2
	title = etf_event_rcw.2.t
	desc = etf_event_rcw.2.d
	picture = BORDER_TENSION_eventPicture

	trigger = {
		always = no
		owner = {
			country_is = { tag = RUS }
			any_war_enemy_country = {
				country_is = { tag = RUS }
			}
		}
		any_core_country = {
			exists = no
			primary_culture = PREV # same as province
			NOT = { has_country_flag = tried_breaking_free_from_russia }
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	is_mtth_scaled_to_size = 1

	immediate = {
		hidden_effect = {
			random_core_country = {
				limit = {
					exists = no
					primary_culture = PREV # same as province
					NOT = { has_country_flag = tried_breaking_free_from_russia }
				}
				save_event_target_as = breakaway_country
				set_country_flag = tried_breaking_free_from_russia
			}
		}
	}

	option = {
		name = etf_event_rcw.2.ra
		owner = {
			start_civil_war = {
				tag = event_target:breakaway_country
				type = attacker
				
				treasury = 150
				manpower = 15
				
				religion = capital
				government = republic
				random_capital = yes
			}
		}
	}

	option = {
		name = etf_event_rcw.2.rb
		trigger = {
			ai = no
		}
		owner = {
			start_civil_war = {
				tag = event_target:breakaway_country
				type = attacker
				play_as = yes
				
				treasury = 150
				manpower = 15
				
				religion = capital
				government = republic
				random_capital = yes
			}
		}
		ai_chance = {
			factor = 0
		}
	}
}

# Civil War Victory
country_event = {
	id = etf_event_rcw.10
	title = etf_event_rcw.10.t
	desc = {
		trigger = {
			is_communist = yes
		}
		desc = etf_event_rcw.10.d1
	}
	desc = {
		trigger = {
			NOT = { is_communist = yes }
		}
		desc = etf_event_rcw.10.d2
	}
	picture = LAND_MILITARY_eventPicture
	major = yes
	major_trigger = {
		OR = {
			same_continent = FROM
			is_neighbor_of = FROM
		}
	}

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		has_global_flag = russian_civil_war_flag
	}

	option = { # Lenin knows best
		name = etf_event_rcw.10.ra
		trigger = {
			is_communist = yes
			has_ruler = "Vladimir Lenin"
		}
		change_government_to_dictatorial_republic = yes
		add_government_reform = communist_government
	}
	option = { # All power to the Soviets
		name = etf_event_rcw.10.rb
		trigger = {
			is_communist = yes
		}
		change_government_to_oligarchic_republic = yes
		add_government_reform = communist_government
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				has_ruler = "Vladimir Lenin"
			}
		}
	}
	option = { # Power to the peasant
		name = etf_event_rcw.10.rc
		trigger = {
			is_communist = yes
		}
		change_government_to_democratic_republic = yes
		add_faction_influence = {
			faction = communist_party
			influence = 50
		}
		ai_chance = {
			factor = 0
		}
	}

	option = { # Long live the Tsar!
		name = etf_event_rcw.10.rd
		trigger = {
			NOT = { is_communist = yes }
		}
		change_government_to_autocratic_monarchy = yes
		set_ruler = former_russian_ruler
		hidden_effect = {
			if = {
				limit = { has_country_flag = former_russian_heir_in_exile }
				set_heir = former_russian_heir
				clr_country_flag = former_russian_heir_in_exile
			}
		}
		ai_chance = {
			factor = 0
		}
	}
	option = { # Kerensky will lead the federation
		name = etf_event_rcw.10.re
		trigger = {
			NOT = { is_communist = yes }
		}
		change_government_to_democratic_republic = yes
		hidden_effect = {
			define_ruler = {
				name = "Alexander Kerensky"
			}
		}
		ai_chance = {
			factor = 0
		}
	}
	option = { # Uh, a compromise?
		name = etf_event_rcw.10.rf
		trigger = {
			NOT = { is_communist = yes }
		}
		change_government_to_parliamentary_monarchy = yes
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_reform = ceremonial_monarch }
				}
				set_ruler = former_russian_ruler
			}
			else = {
				define_ruler = {}
			}
		}
	}
}

# Finnish Independence
country_event = {
	id = etf_event_rcw.20
	title = etf_event_rcw.20.t
	desc = etf_event_rcw.20.d
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes

	trigger = {
		is_subject_of_type = governorate_subject
		country_is = { tag = FIN }
		overlord = {
			country_is = { tag = RUS }
			NOT = { government = monarchy }
		}
	}

	option = { # Finland marches ahead, alone!
		name = etf_event_rcw.20.ra
		overlord = {
			country_event = { id = etf_event_rcw.21 }
		}
		hidden_effect = {
			every_province = {
				limit = {
					OR = {
						province_id = 19
						province_id = 27
						province_id = 28
						province_id = 29
						province_id = 30
						province_id = 31
						province_id = 32
						province_id = 4113
						province_id = 4123
						province_id = 4124
						province_id = 4151
						province_id = 4152
						province_id = 1930
					}
					NOT = { is_core = ROOT }
				}
				add_core = ROOT
			}
			1930 = { # Åland
				if = {
					limit = {
						owner = {
							overlord_of = ROOT
						}
					}
					cede_province = ROOT
				}
			}
		}
		grant_independence = yes
		change_government_to_democratic_republic = yes
	}

	option = { # We're content
		name = etf_event_rcw.20.rb
		add_stability = -1
		ai_chance = {
			factor = 0
		}
	}
}
